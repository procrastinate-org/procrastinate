248def defer( 

249 app: procrastinate.App, 

250 task: str, 

251 json_args: str, 

252 lock: Optional[str], 

253 queueing_lock: Optional[str], 

254 ignore_already_enqueued: bool, 

255 queue: Optional[str], 

256 at: Optional[str], 

257 in_: Optional[int], 

258 unknown: bool, 

259): 

260 """ 

261 Create a job from the given task, to be executed by a worker. 

262 TASK should be the name or dotted path to a task. 

263 JSON_ARGS should be a json object (a.k.a dictionary) with the job parameters 

264 """ 

265 # Loading json args 

266 args = load_json_args( 

267 json_args=json_args, json_loads=app.connector.json_loads or json.loads 

268 ) 

269 

270 if at is not None and in_ is not None: 

271 raise click.UsageError("Cannot use both --at and --in") 

272 

273 schedule_at = get_schedule_at(at=at) 

274 schedule_in = get_schedule_in(in_=in_) 

275 

276 # Build kwargs. Remove all None kwargs to use their default values. 

277 configure_kwargs = { 

278 "lock": lock, 

279 "queueing_lock": queueing_lock, 

280 "schedule_at": schedule_at, 

281 "schedule_in": schedule_in, 

282 "queue": queue, 

283 } 

284 configure_kwargs = filter_none(configure_kwargs) 

285 

286 # Configure the job. If the task is known, it will be used. 

287 job_deferrer = configure_task( 

288 app=app, 

289 task_name=task, 

290 configure_kwargs=configure_kwargs, 

291 allow_unknown=unknown, 

292 ) 

293 

294 # Printing info 

295 str_kwargs = ", ".join(f"{k}={repr(v)}" for k, v in args.items()) 

296 click.echo(f"Launching a job: {task}({str_kwargs})") 

297 

298 # And launching the job 

299 try: 

300 job_deferrer.defer(**args) # type: ignore 

301 except exceptions.AlreadyEnqueued as exc: 

302 if not ignore_already_enqueued: 

303 raise 

304 click.echo(f"{exc} (ignored)")

308 return {key: value for key, value in dictionary.items() if value is not None}

321 )
