112class PeriodicDeferrer: 

113 def __init__( 

114 self, 

115 registry: PeriodicRegistry, 

116 max_delay: float = MAX_DELAY, 

117 ): 

118 # {(task_name, periodic_id): defer_timestamp} 

119 self.last_defers: dict[tuple[str, str], int] = {} 

120 self.max_delay = max_delay 

121 self.registry = registry

123 async def worker(self) -> None: 

124 """ 

125 High-level command for the periodic deferrer. Launches the loop. 

126 """ 

127 if not self.registry.periodic_tasks: 

128 logger.info( 

129 "No periodic task found, periodic deferrer will not run.", 

130 extra={"action": "periodic_deferrer_no_task"}, 

131 ) 

132 return 

133 

134 while True: 

135 now = time.time() 

136 await self.defer_jobs(jobs_to_defer=self.get_previous_tasks(at=now)) 

137 await self.wait(next_tick=self.get_next_tick(at=now))

139 def get_next_tick(self, at: float): 

140 """ 

141 Return the number of seconds to wait before the next periodic task needs to be 

142 deferred. 

143 If now is not passed, the current timestamp is used. 

144 """ 

145 next_timestamp = min( 

146 pt.croniter.get_next(ret_type=float, start_time=at) # type: ignore 

147 for pt in self.registry.periodic_tasks.values() 

148 ) 

149 return next_timestamp - at 

150 

151 def get_previous_tasks(self, at: float) -> Iterable[TaskAtTime]: 

152 """ 

153 Return each periodic task that may not have been deferred yet, alongside with 

154 its scheduled time. 

155 Tasks that should have been deferred more than self.max_delay seconds ago are 

156 ignored. 

157 """ 

158 for key, periodic_task in self.registry.periodic_tasks.items(): 

159 for timestamp in self.get_timestamps( 

160 periodic_task=periodic_task, 

161 since=self.last_defers.get(key), 

162 until=at, 

163 ): 

164 self.last_defers[key] = timestamp 

165 yield periodic_task, timestamp 

166 

167 def get_timestamps( 

168 self, periodic_task: PeriodicTask, since: int | None, until: float 

169 ) -> Iterable[int]: 

170 cron_iterator = periodic_task.croniter 

171 if since: 

172 timestamp = cron_iterator.set_current(start_time=since) 

173 while True: 

174 timestamp = round(cron_iterator.get_next(ret_type=float)) 

175 if timestamp > until: 

176 return 

177 yield int(timestamp) 

178 

179 else: 

180 cron_iterator.set_current(start_time=until) 

181 timestamp = round(cron_iterator.get_prev(ret_type=float)) 

182 delay = until - timestamp 

183 

184 if delay > self.max_delay: 

185 logger.debug( 

186 f"Ignoring periodic task {periodic_task.task.name} scheduled " 

187 f"more than {self.max_delay} s ago ({delay:.0f} s)", 

188 extra={ 

189 "task": periodic_task.task.name, 

190 "periodic_id": periodic_task.periodic_id, 

191 "action": "ignore_periodic_task", 

192 "max_delay": self.max_delay, 

193 "delay": delay, 

194 }, 

195 ) 

196 return 

197 

198 yield int(timestamp)

200 async def defer_jobs(self, jobs_to_defer: Iterable[TaskAtTime]) -> None: 

201 """ 

202 Try deferring all tasks that might need deferring. The database will keep us 

203 from deferring the same task for the same scheduled time multiple times. 

204 """ 

205 for periodic_task, timestamp in jobs_to_defer: 

206 task = periodic_task.task 

207 periodic_id = periodic_task.periodic_id 

208 configure_kwargs = periodic_task.configure_kwargs 

209 task_kwargs = configure_kwargs.get("task_kwargs") 

210 if task_kwargs is None: 

211 task_kwargs = {} 

212 configure_kwargs["task_kwargs"] = task_kwargs 

213 task_kwargs["timestamp"] = timestamp 

214 

215 description = { 

216 "task_name": task.name, 

217 "periodic_id": periodic_id, 

218 "defer_timestamp": timestamp, 

219 "kwargs": configure_kwargs, 

220 } 

221 job_deferrer = task.configure(**configure_kwargs) 

222 job = job_deferrer.job 

223 try: 

224 job_id = await job_deferrer.job_manager.defer_periodic_job( 

225 job=job, 

226 periodic_id=periodic_id, 

227 defer_timestamp=timestamp, 

228 ) 

229 except exceptions.AlreadyEnqueued: 

230 logger.debug( 

231 f"Periodic job {job.call_string} cannot be enqueued: there is already " 

232 f"a job in the queue with the queueing lock {task.queueing_lock}", 

233 extra={ 

234 "action": "skip_periodic_task_queueing_lock", 

235 "queueing_lock": task.queueing_lock, 

236 "job": job.log_context(), 

237 **description, 

238 }, 

239 ) 

240 continue 

241 else: 

242 job = job.evolve(id=job_id) 

243 

244 if job.id: 

245 logger.info( 

246 f"Periodic job {job.call_string} deferred for timestamp " 

247 f"{timestamp} with id {job.id}", 

248 extra={ 

249 "action": "periodic_task_deferred", 

250 "job": job.log_context(), 

251 **description, 

252 }, 

253 ) 

254 else: 

255 logger.debug( 

256 f"Periodic job {job.call_string} skipped for timestamp " 

257 f"{timestamp}: already deferred", 

258 extra={ 

259 "action": "periodic_task_already_deferred", 

260 "job": job.log_context(), 

261 **description, 

262 }, 

263 )
