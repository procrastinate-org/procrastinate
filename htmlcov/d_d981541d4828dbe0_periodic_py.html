14# The maximum delay after which tasks will be considered as

48 def periodic_decorator(self, cron: str, periodic_id: str, **kwargs): 

49 """ 

50 Decorator over a task definition that registers that task for periodic

52 to be used instead. 

53 """ 

54

63 async def worker(self) -> None: 

64 """ 

65 High-level command for the periodic deferrer. Launches the loop. 

66 """ 

67 if not self.periodic_tasks: 

68 logger.info( 

69 "No periodic task found, periodic deferrer will not run.", 

70 extra={"action": "periodic_deferrer_no_task"}, 

71 ) 

72 return 

73 

74 while True: 

75 now = time.time() 

76 await self.defer_jobs(jobs_to_defer=self.get_previous_tasks(at=now)) 

77 await self.wait(next_tick=self.get_next_tick(at=now))

81 def register_task( 

82 self, 

83 task: tasks.Task, 

84 cron: str, 

85 periodic_id: str, 

86 configure_kwargs: dict[str, Any], 

87 ) -> PeriodicTask: 

88 key = (task.name, periodic_id) 

89 if key in self.periodic_tasks: 

90 raise exceptions.TaskAlreadyRegistered( 

91 "A periodic task was already registed with the same periodic_id " 

92 f"({periodic_id!r}). Please use a different periodic_id for multiple " 

93 "schedules of the same task." 

94 ) 

95 

96 logger.info( 

97 f"Registering task {task.name} with periodic id {periodic_id!r} to run " 

98 f"periodically with cron {cron}", 

99 extra={ 

100 "action": "registering_periodic_task", 

101 "task": task.name, 

102 "cron": cron, 

103 "periodic_id": periodic_id, 

104 "kwargs": str(configure_kwargs), 

105 }, 

106 ) 

107 

108 self.periodic_tasks[key] = periodic_task = PeriodicTask( 

109 task=task, 

110 cron=cron, 

111 periodic_id=periodic_id, 

112 configure_kwargs=configure_kwargs, 

113 ) 

114 return periodic_task

116 def get_next_tick(self, at: float): 

117 """ 

118 Return the number of seconds to wait before the next periodic task needs to be 

119 deferred. 

120 If now is not passed, the current timestamp is used. 

121 """ 

122 next_timestamp = min( 

123 pt.croniter.get_next(ret_type=float, start_time=at) # type: ignore 

124 for pt in self.periodic_tasks.values() 

125 ) 

126 return next_timestamp - at

128 def get_previous_tasks(self, at: float) -> Iterable[TaskAtTime]: 

129 """ 

130 Return each periodic task that may not have been deferred yet, alongside with 

131 its scheduled time. 

132 Tasks that should have been deferred more than self.max_delay seconds ago are 

133 ignored. 

134 """ 

135 for key, periodic_task in self.periodic_tasks.items(): 

136 for timestamp in self.get_timestamps( 

137 periodic_task=periodic_task, 

138 since=self.last_defers.get(key), 

139 until=at, 

140 ): 

141 self.last_defers[key] = timestamp 

142 yield periodic_task, timestamp 

143 

144 def get_timestamps( 

145 self, periodic_task: PeriodicTask, since: int | None, until: float 

146 ) -> Iterable[int]: 

147 cron_iterator = periodic_task.croniter 

148 if since: 

149 timestamp = cron_iterator.set_current(start_time=since) 

150 while True: 

151 timestamp = round(cron_iterator.get_next(ret_type=float)) 

152 if timestamp > until: 

153 return 

154 yield int(timestamp) 

155 

156 else: 

157 cron_iterator.set_current(start_time=until) 

158 timestamp = round(cron_iterator.get_prev(ret_type=float)) 

159 delay = until - timestamp 

160 

161 if delay > self.max_delay: 

162 logger.debug( 

163 f"Ignoring periodic task {periodic_task.task.name} scheduled " 

164 f"more than {self.max_delay} s ago ({delay:.0f} s)", 

165 extra={ 

166 "task": periodic_task.task.name, 

167 "periodic_id": periodic_task.periodic_id, 

168 "action": "ignore_periodic_task", 

169 "max_delay": self.max_delay, 

170 "delay": delay, 

171 }, 

172 ) 

173 return 

174 

175 yield int(timestamp)

177 async def defer_jobs(self, jobs_to_defer: Iterable[TaskAtTime]) -> None: 

178 """ 

179 Try deferring all tasks that might need deferring. The database will keep us 

180 from deferring the same task for the same scheduled time multiple times. 

181 """ 

182 for periodic_task, timestamp in jobs_to_defer: 

183 task = periodic_task.task 

184 periodic_id = periodic_task.periodic_id 

185 configure_kwargs = periodic_task.configure_kwargs 

186 configure_kwargs.setdefault("task_kwargs", {})["timestamp"] = timestamp 

187 description = { 

188 "task_name": task.name, 

189 "periodic_id": periodic_id, 

190 "defer_timestamp": timestamp, 

191 "kwargs": configure_kwargs, 

192 } 

193 job_deferrer = task.configure(**configure_kwargs) 

194 job = job_deferrer.job 

195 try: 

196 job_id = await job_deferrer.job_manager.defer_periodic_job( 

197 job=job, 

198 periodic_id=periodic_id, 

199 defer_timestamp=timestamp, 

200 ) 

201 except exceptions.AlreadyEnqueued: 

202 logger.debug( 

203 f"Periodic job {job.call_string} cannot be enqueued: there is already " 

204 f"a job in the queue with the queueing lock {task.queueing_lock}", 

205 extra={ 

206 "action": "skip_periodic_task_queueing_lock", 

207 "queueing_lock": task.queueing_lock, 

208 "job": job.log_context(), 

209 **description, 

210 }, 

211 ) 

212 continue 

213 else: 

214 job = job.evolve(id=job_id) 

215 

216 if job.id: 

217 logger.info( 

218 f"Periodic job {job.call_string} deferred for timestamp " 

219 f"{timestamp} with id {job.id}", 

220 extra={ 

221 "action": "periodic_task_deferred", 

222 "job": job.log_context(), 

223 **description, 

224 }, 

225 ) 

226 else: 

227 logger.debug( 

228 f"Periodic job {job.call_string} skipped for timestamp " 

229 f"{timestamp}: already deferred", 

230 extra={ 

231 "action": "periodic_task_already_deferred", 

232 "job": job.log_context(), 

233 **description, 

234 }, 

235 )
