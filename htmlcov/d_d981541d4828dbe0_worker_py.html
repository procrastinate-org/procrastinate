88 def context_for_worker( 

89 self, worker_id: int, reset=False, **kwargs 

90 ) -> job_context.JobContext: 

91 """ 

92 Retrieves the context for sub-sworker ``worker_id``. If not found, or ``reset`` 

93 is True, context is recreated from ``self.base_context``. Additionnal parameters 

94 are used to update the context. The resulting context is kept and will be 

95 returned for later calls. 

96 """ 

97 if reset or worker_id not in self.current_contexts:
